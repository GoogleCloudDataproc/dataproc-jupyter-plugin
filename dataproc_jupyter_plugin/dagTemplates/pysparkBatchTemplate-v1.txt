def write_output_to_file(run_id, **kwargs):
    output_file_path = f"{{output_notebook}}{run_id}.ipynb"
    print(output_file_path)
    kwargs['ti'].xcom_push(key='output_file_path', value=output_file_path)
    return output_file_path
